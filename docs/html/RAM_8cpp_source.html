71void RAM::loadProgram(const string &filename) {
72 if (!isReset_) reset_();
73 ifstream file(filename);
74 try {
75 executable_ = true;
76 if (file.is_open()) {
77 program_.clear();
78 string line;
79 bool data = false, comment = false;
80 unsigned memory_ptr = 0, stringMemory_ptr = 0;
81
82 while (getline(file, line)) {
83 if (line.empty()) {
84 continue;
85 }
86
87 if (line.find("/*") != string::npos) {
88 comment = true;
89 } else if (line.find("*/") != string::npos) {
90 comment = false;
91 }
92
93 if (!comment) {
94 if (line[0] != '/' && line[1] != '/') {
95 if (line.find("/*") != string::npos) {
96 comment = true;
97 continue;
98 } else if (line.find("*/") != string::npos) {
99 comment = false;
100 continue;
101 }
102 if (line.find(".data") != string::npos) {
103 data = true;
104 continue;
105 } else if (line.find(".code") != string::npos) {
106 data = false;
107 continue;
108 }
109
110 if (line.find(':') != string::npos) {
111 if (!data)
112 labels_[line.substr(0, line.find(':'))] =
113 pair<unsigned, bool>(program_.size(), false);
114 else
115 try {
116 createData_(line, memory_ptr, stringMemory_ptr);
117 } catch (const exception &e) {
118 executable_ = false;
119 throw invalid_argument("Error creating data: " +
120 string(e.what()));
121 }
122 } else {
123 try {
124 if (!data) createInstruction_(line);
125 } catch (const exception &e) {
126 executable_ = false;
127 throw invalid_argument("Error creating instruction: " +
128 string(e.what()));
129 }
130 }
131 }
132 }
133 }
134
135 if (comment) {
136 executable_ = false;
137 throw invalid_argument("The program has an unclosed comment");
138 }
139 if (program_.empty()) {
140 executable_ = false;
141 throw invalid_argument("No instructions found in file: " + filename);
142 }
143
144 file.close();
145 cout << "Program \"" << filename << "\" loaded successfully" << endl;
146 commandHistory_.push_back("Load program: " + filename + " (success)");
147 } else {
148 executable_ = false;
149 throw invalid_argument("Error opening file: " + filename);
150 }
151 } catch (const exception &e) {
152 cout << "Error loading program: " << e.what() << endl;
153 file.close();
154 commandHistory_.push_back("Load program: " + filename + " (error)");
155 }
156 isReset_ = false;
157}

233void RAM::createData_(const string &line, unsigned &memory_ptr,
234 unsigned &stringMemory_ptr) {
235 if (line.empty()) {
236 return;
237 }
238
239 if (line.find(".string") != string::npos) { // tag: .string "string"
240 if (stringMemory_ptr >= numStringMemory_) {
241 throw invalid_argument("String memory limit exceeded");
242 }
243 stringMemory_.push_back(
244 line.substr(line.find('"') + 1, line.rfind('"') - line.find('"') - 1));
245 labels_[line.substr(0, line.find(':'))] =
246 pair<unsigned, bool>(stringMemory_ptr, true);
247 stringMemory_ptr++;
248 } else if (line.find(".int") != string::npos) { // tag: .int 123
249 if (memory_ptr >= numMemory_) {
250 throw invalid_argument("Memory limit exceeded");
251 }
252 memory_.push_back(stoi(line.substr(line.find('.') + 5)));
253 labels_[line.substr(0, line.find(':'))] =
254 pair<unsigned, bool>(memory_ptr, false);
255 memory_ptr++;
256 } else if (line.find(".float") != string::npos) { // tag: .float 123.45
257 if (memory_ptr >= numMemory_) {
258 throw invalid_argument("Memory limit exceeded");
259 }
260 memory_.push_back(stof(line.substr(line.find('.') + 7)));
261 labels_[line.substr(0, line.find(':'))] =
262 pair<unsigned, bool>(memory_ptr, false);
263 memory_ptr++;
264 } else if (line.find(".double") != string::npos) { // tag: .double 123.45
265 if (memory_ptr >= numMemory_) {
266 throw invalid_argument("Memory limit exceeded");
267 }
268 memory_.push_back(stod(line.substr(line.find('.') + 8)));
269 labels_[line.substr(0, line.find(':'))] =
270 pair<unsigned, bool>(memory_ptr, false);
271 memory_ptr++;
272 } else {
273 throw invalid_argument("Unknown data type: " + line);
274 }
