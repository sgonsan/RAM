49void RAMMachine::loadProgram(const string &filename) {
50 reset_();
51 ifstream file(filename);
52 try {
53 executable_ = true;
54 if (file.is_open()) {
55 program_.clear();
56 string line;
57 bool data = false, comment = false;
58 unsigned memory_ptr = 0, stringMemory_ptr = 0;
59
60 while (getline(file, line)) {
61 if (line.empty()) {
62 continue;
63 }
64
65 if (line.find("/*") != string::npos) {
66 comment = true;
67 } else if (line.find("*/") != string::npos) {
68 comment = false;
69 }
70
71 if (!comment) {
72 if (line[0] != '/' && line[1] != '/') {
73 if (line.find("/*") != string::npos) {
74 comment = true;
75 continue;
76 } else if (line.find("*/") != string::npos) {
77 comment = false;
78 continue;
79 }
80 if (line.find(".data") != string::npos) {
81 data = true;
82 continue;
83 } else if (line.find(".code") != string::npos) {
84 data = false;
85 continue;
86 }
87
88 if (line.find(':') != string::npos) {
89 if (!data)
90 labels_[line.substr(0, line.find(':'))] =
91 pair<unsigned, bool>(program_.size(), false);
92 else
93 try {
94 createData(line, memory_ptr, stringMemory_ptr);
95 } catch (const exception &e) {
96 executable_ = false;
97 throw invalid_argument("Error creating data: " +
98 string(e.what()));
99 }
100 } else {
101 try {
102 if (!data) createInstruction(line);
103 } catch (const exception &e) {
104 executable_ = false;
105 throw invalid_argument("Error creating instruction: " +
106 string(e.what()));
107 }
108 }
109 }
110 }
111 }
112
113 if (comment) {
114 executable_ = false;
115 throw invalid_argument("The program has an unclosed comment");
116 }
117 if (program_.empty()) {
118 executable_ = false;
119 throw invalid_argument("No instructions found in file: " + filename);
120 }
121
122 file.close();
123 cout << "Program \"" << filename << "\" loaded successfully" << endl;
124 } else {
125 executable_ = false;
126 throw invalid_argument("Error opening file: " + filename);
127 }
128 } catch (const exception &e) {
129 cout << "Error loading program: " << e.what() << endl;
130 file.close();
131 }
132}

202void RAMMachine::createData(const string &line, unsigned &memory_ptr,
203 unsigned &stringMemory_ptr) {
204 if (line.empty()) {
205 return;
206 }
207
208 if (line.find(".string") != string::npos) { // tag: .string "string"
209 stringMemory_.push_back(
210 line.substr(line.find('"') + 1, line.rfind('"') - line.find('"') - 1));
211 labels_[line.substr(0, line.find(':'))] =
212 pair<unsigned, bool>(stringMemory_ptr, true);
213 stringMemory_ptr++;
214 } else if (line.find(".int") != string::npos) { // tag: .int 123
215 memory_.push_back(stoi(line.substr(line.find('.') + 5)));
216 labels_[line.substr(0, line.find(':'))] =
217 pair<unsigned, bool>(memory_ptr, false);
218 memory_ptr++;
219 } else if (line.find(".float") != string::npos) { // tag: .float 123.45
220 memory_.push_back(stof(line.substr(line.find('.') + 7)));
221 labels_[line.substr(0, line.find(':'))] =
222 pair<unsigned, bool>(memory_ptr, false);
223 memory_ptr++;
224 } else if (line.find(".double") != string::npos) { // tag: .double 123.45
225 memory_.push_back(stod(line.substr(line.find('.') + 8)));
226 labels_[line.substr(0, line.find(':'))] =
227 pair<unsigned, bool>(memory_ptr, false);
228 memory_ptr++;
229 } else {
230 throw invalid_argument("Unknown data type: " + line);
